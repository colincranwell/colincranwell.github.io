from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt

# Normalize the data
scaler = StandardScaler()
normalized_data = scaler.fit_transform(spotify_data.select_dtypes(include=['float64', 'int64']))

inertia = []
range_clusters = range(1, 11)
for k in range_clusters:
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans.fit(normalized_data)
    inertia.append(kmeans.inertia_)

# Elbow Method Visualization
plt.figure(figsize=(8, 5))
plt.plot(range_clusters, inertia, marker='o', color='teal', label='Inertia')
plt.xticks(range_clusters)
plt.xlabel('Number of Clusters', fontsize=12)
plt.ylabel('Inertia (Sum of Squared Distances)', fontsize=12)
plt.title('Elbow Method: Optimal Number of Clusters', fontsize=14)
plt.grid(color='gray', linestyle='--', linewidth=0.5)
plt.legend()
plt.show()

from sklearn.decomposition import PCA
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans
import numpy as np

kmeans = KMeans(n_clusters=4, random_state=42)
clusters = kmeans.fit_predict(normalized_data) 
centroids = kmeans.cluster_centers_

print("Centroid Values for Each Cluster:")
for idx, centroid in enumerate(centroids):
    print(f"Cluster {idx + 1}: {centroid}")

cluster_descriptions = []

pca = PCA(n_components=2)
pca_result = pca.fit_transform(normalized_data)

plt.figure(figsize=(10, 6))
scatter = plt.scatter(pca_result[:, 0], pca_result[:, 1], c=clusters, cmap='viridis', s=50, alpha=0.7)

centroids_pca = pca.transform(kmeans.cluster_centers_)
for idx, (x, y) in enumerate(centroids_pca):
    plt.scatter(x, y, color='red', marker='X', s=200, edgecolor='black')

# Add labels and title
plt.xlabel('PCA Component 1')
plt.ylabel('PCA Component 2')
plt.title('Spotify Clusters Visualized with PCA')
plt.colorbar(scatter, label='Cluster')
plt.grid(alpha=0.3)

# Show the plot
plt.show()
